# higher-education-school
全国高等学校列表


## 安装

`npm install higher-education-school`

## 用法

```

var hes = require('higher-education-school');

var ynu = hes[1996];
// ynu['学校名称'] === '云南大学'
// ynu['主管部门'] === '云南省'
// ynu['所在地'] === '昆明市'
// ynu['所在省市'] === '云南省'
// ynu['办学层次'] === '本科'
// ynu['备注'] === ''

```
